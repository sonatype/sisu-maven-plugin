167     public void execute()
168         throws MojoExecutionException
169     {
170         final List<URL> urls = new ArrayList<URL>();
171         appendToClassPath( urls, outputDirectory );
172         if ( includeDependencies )
173         {
174             final FilterArtifacts filter = new FilterArtifacts();
175 
176             filter.addFilter( new ProjectTransitivityFilter( project.getDependencyArtifacts(), excludeTransitive ) );
177             filter.addFilter( new ScopeFilter( cleanList( includeScope ), cleanList( excludeScope ) ) );
178             filter.addFilter( new TypeFilter( cleanList( includeTypes ), cleanList( excludeTypes ) ) );
179             filter.addFilter( new ClassifierFilter( cleanList( includeClassifiers ), cleanList( excludeClassifiers ) ) );
180             filter.addFilter( new GroupIdFilter( cleanList( includeGroupIds ), cleanList( excludeGroupIds ) ) );
181             filter.addFilter( new ArtifactIdFilter( cleanList( includeArtifactIds ), cleanList( excludeArtifactIds ) ) );
182 
183             try
184             {
185                 for ( final Object artifact : filter.filter( project.getArtifacts() ) )
186                 {
187                     appendToClassPath( urls, ( (Artifact) artifact ).getFile() );
188                 }
189             }
190             catch ( final ArtifactFilterException e )
191             {
192                 throw new MojoExecutionException( e.getMessage(), e );
193             }
194         }
195         new SisuIndex( outputDirectory ).index( urls );
196     }
197 
198     // ----------------------------------------------------------------------
199     // Implementation methods
200     // ----------------------------------------------------------------------
201 
202     private void appendToClassPath( final List<URL> urls, final File file )
203     {
204         if ( null != file )
205         {
206             try
207             {
208                 urls.add( file.toURI().toURL() );
209             }
210             catch ( final MalformedURLException e )
211             {
212                 getLog().warn( e.getLocalizedMessage() );
213             }
214         }
215     }
216 
217     private static String cleanList( final String list )
218     {
219         return StringUtils.isEmpty( list ) ? "" : StringUtils.join( StringUtils.split( list ), "," );
220     }
